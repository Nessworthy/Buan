  340.     public function sql() {

  344.         $sql->query = '';
  345.         $sql->bindings = array();
  346.  
  347.         // Generate WHERE portion from the root clause group and all sub-groups

  349.         $whereSql->query = str_replace("()", "", $whereSql->query);
  350.         if($whereSql->query!='') {

  352.             $sql->query = " WHERE {$whereSql->query}";
  353.             $sql->bindings = array_merge($sql->bindings, $whereSql->bindings);
  354.         }
  355.  
  356.         // Add GROUP BYs
  357.         if(!empty($this->groupBys)) {
  358.             $sql->query .= ' GROUP BY '.implode(", ", $this->groupBys);
  359.         }
  360.  
  361.         // Add HAVING clauses
  362.         $havingSql = $this->havingGroup->sql();
  363.         $havingSql->query = str_replace("()", "", $havingSql->query);
  364.         if($havingSql->query!='') {

  366.             $sql->bindings = array_merge($sql->bindings, $havingSql->bindings);
  367.         }
  368.  
  369.         // Add ORDER BYs
  370.         if(count($this->orders)>0) {

  372.             $orders = array();

  374.                 $orders[] = $order['fieldName'].' '.$order['direction'];
  375.             }
  376.             $sql->query .= implode(", ", $orders);
  377.         }
  378.  
  379.         // Add LIMIT
  380.         if(!is_null($this->limit)) {
  381.             $sql->query .= ' LIMIT '.$this->limit['recordCount'].' OFFSET '.$this->limit['start'];
  382.         }
  383.  
  384.         // Add LEFT JOINs
  385.         if(!empty($this->leftJoins)) {
  386.             $joins = array_reverse($this->leftJoins);
  387.             foreach($joins as $join) {
  388.                 $sql->query = ' LEFT JOIN '.$join['table'].' ON '.$join['clause'].' '.$sql->query;
  389.             }
  390.         }
  391.  
  392.         // Add SELECT fields
  393.         if(!empty($this->selects['fields'])) {
  394.             $fields = array();
  395.             foreach($this->selects['fields'] as $field) {
  396.                 $fields[] = "{$field->query} ".($field->alias===NULL ? '' : " AS {$field->alias}");
  397.                 if($field->bindings!==NULL) {
  398.                     $sql->bindings = array_merge($sql->bindings, $field->bindings);
  399.                 }
  400.             }
  401.             $sql->query = 'SELECT '.implode(", ", $fields).' '.(!empty($this->selects['tables']) ? 'FROM '.implode(", ", $this->selects['tables']).' ' : '').$sql->query;
  402.         }
  403.  
  404.         // Result
  405.         return $sql;
  406.     }
