  375.                 continue;

  381.                 if(!isset($model->{$foreignKey}) || $model->{$foreignKey}<=-1) {
  382.                     $model->{$foreignKey} = $rModel->getPrimaryKeyValue();

  384.                 continue;

  386.  
  387.             // Save and set foreign-key

  389.             else if($rModel->getModelManager()->save($rModel)) {
  390.                 if(!isset($model->{$foreignKey}) || $model->{$foreignKey}<=-1) {
  391.                     $model->{$foreignKey} = $rModel->getPrimaryKeyValue();
  392.                 }
  393.             }
  394.  
  395.             // On save failure, rollback entire transaction and return

  401.                     SystemLog::add($e->getMessage(), SystemLog::WARNING);

  524.         if(!$model->hasChanged()) {
