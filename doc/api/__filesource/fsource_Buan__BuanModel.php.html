  419.     /*
  420.     # @method void addRelatedModel( Model $model, [string $relationRef, [string $reverseRelationRef, [bool $cancelBubble]]] )
  421.     # $model                = The Model that will become related to $this Model
  422.     # $relationRef            = The specific relationship reference that we'll use when adding $model (only required if several native keys point to same Model type)
  423.     # $reverseRelationRef    = The inverse of $relationRef (REQUIRED only for recursive M:M relationships)
  424.     # $cancelBubble            = Used to prevent recursion (INTERNAL USE ONLY)
  425.     #
  426.     # Store the $model in the appropriate $this->relatedModels array.
  427.     # Thanks to $this->__call(), you can also call more user-friendly methods
  428.     # which actually get passed through to this method.
  429.     # eg.    $A->addBlogEntry($entry) ... passes through to ... $A->addRelatedModel($entry);
  430.     #
  431.     # Duplicate associations are prevented, except in the case of M:M linking
  432.     # Models.
  433.     #
  434.     # There is an important note to make when it comes to adding related Models in a
  435.     # recursive 1:M relationship ... in this method, unless $relationRef is specified then 
  436.     # $model is ALWAYS considered the "M" side of the relationship and $this is the "1" side.
  437.     # ie.    calling "$A->addRelatedModel($B)" will add $B to $A->relatedModels[1:M]
  438.     */
  439.     public function addRelatedModel($model, $relationRef=ModelRelation::REF_DEFAULT, $reverseRelationRef=NULL, $cancelBubble=FALSE) {
  440.  
  441.         // Determine the relationship between $this and $model
  442.         $relation = ModelRelation::getRelation($this->modelName, $model->modelName, $relationRef);
  443.  
  444.         if($reverseRelationRef!==NULL) {
  445.             $reverseRelation = ModelRelation::getRelation($this->modelName, $model->modelName, $reverseRelationRef);
  446.         }
  447.  
  448.         // If the relationship is M:M, we need to create an instance of the
  449.         // appropriate linking Model and associate $this and $model to that
  450.         // instance.
  451.         if($relation->isManyToMany()) {
  452.  
  453.             // First, check if we've already got a linking Model instance that contains the two instances, $this and $model
  454.             $linkModels = $this->getRelatedModels($relation->getLinkModel(), $relation->getReference());
  455.             foreach($linkModels as $linkModel) {
  456.                 $foreignModel = $linkModel->getRelatedModelsByCardinality(ModelRelation::MANY_TO_ONE);
  457.                 $foreignKey = $linkModel->getForeignKey($model, $relation->getReference());
  458.                 if(isset($foreignModel[$foreignKey]) && $foreignModel[$foreignKey]===$model) {
  459.                     // Already present, don't add anything
  460.                     return NULL;
  461.                 }
  462.             }
  463.  
  464.             // Create a new instance of the linking Model and associate it with $this and $model
  465.             $linkModel = Model::create($relation->getLinkModel());
  466.             if($relation->isRecursive()) {
  467.  
  468.                 // If no relation references have been specified, then we first
  469.                 // attempt to use the REF_PARENT/REF_CHILD pairing.
  470.                 // If these fail then stop.
  471.                 if($reverseRelationRef===NULL) {
  472.                     $reverseRelationRef = $relationRef===ModelRelation::REF_PARENT ? ModelRelation::REF_CHILD : ($relationRef===ModelRelation::REF_CHILD ? ModelRelation::REF_PARENT : NULL);
  473.                     if($reverseRelationRef===NULL) {
  474.                         SystemLog::add('When adding a related Model via a M:M relationship, you must specify the reverse relation reference (usually REF_PARENT or REF_CHILD).', SystemLog::WARNING);
  475.                         return NULL;
  476.                     }
  477.                 }
  478.                 $this->addRelatedModel($linkModel, $relationRef);
  479.                 $linkModel->addRelatedModel($model, $reverseRelationRef);
  480.             }
  481.             else {
  482.                 $linkModel->addRelatedModel($this, $relation->getReference());
  483.                 $linkModel->addRelatedModel($model, $relation->getReference());
  484.             }
  485.             return NULL;
  486.         }
  487.  
  488.         // Deal with recurive relationships
  489.         // Here we do various checks to make sure that both the 1:M and M:1
  490.         // entries in $this->relatedModels are populated instead of recursively
  491.         // populating 1:M, 1:M, 1:M, etc.
  492.         if($relation->isRecursive()) {
  493.  
  494.             // 1:M
  495.             if($relation->isOneToMany()) {
  496.                 $foreignModels = $model->getRelatedModelsByCardinality(ModelRelation::ONE_TO_MANY);
  497.                 if(isset($foreignModels[$this->modelName][$relation->getReference()]) && in_array($this, $foreignModels[$this->modelName][$relation->getReference()], TRUE)) {
  498.                     $relation = $relation->getInverseRelation();
  499.                 }
  500.             }
  501.  
  502.             // M:1
  503.             else if($relation->isManyToOne()) {
  504.                 $foreignModels = $model->getRelatedModelsByCardinality(ModelRelation::MANY_TO_ONE);
  505.                 $fKey = $this->getForeignKey($model);
  506.                 if(isset($foreignModels[$fKey]) && $foreignModels[$fKey]===$this) {
  507.                     $relation = $relation->getInverseRelation();
  508.                 }
  509.             }
  510.         }
  511.  
  512.         // M:1
  513.         if($relation->isManyToOne()) {
  514.  
  515.             // Create/update entry in $this->relatedModels
  516.             $cardinality = $relation->getCardinality();
  517.             $ref = $relation->getReference();
  518.             $foreignKey = $this->getForeignKey($model, $ref);
  519.             if(isset($this->relatedModels[$cardinality][$foreignKey])) {
  520.                 if($this->relatedModels[$cardinality][$foreignKey]!==$model) {
  521.                     // First, remove $this from the Model to which it's foreign-key is currently pointing
  522.                     $this->relatedModels[$cardinality][$foreignKey]->removeRelatedModel($this);
  523.                 }
  524.                 else {
  525.                     // Don't need to do anything as $model is already related to $this
  526.                     return NULL;
  527.                 }
  528.             }
  529.             $this->relatedModels[$cardinality][$foreignKey] = $model;
  530.             if(!$cancelBubble) {
  531.                 $model->addRelatedModel($this, $relationRef, $reverseRelationRef, TRUE);
  532.             }
  533.         }
  534.  
  535.         // 1:M (and 1:1)
  536.         if($relation->isOneToMany()) {
  537.  
  538.             // Create an entry for the type of $model in the 1:M relations
  539.             // array, if none exists
  540.             $cardinality = $relation->getCardinality();
  541.             $ref = $relation->getReference();
  542.             if(!isset($this->relatedModels[$cardinality][$model->modelName][$ref])) {
  543.                 $this->relatedModels[$cardinality][$model->modelName][$ref] = array();
  544.             }
  545.  
  546.             // Add $model to the 1:M relations array (if it isn't already)
  547.             $alreadyInArray = FALSE;
  548.             if(!$model->isInDatabase()) {
  549.                 $alreadyInArray = in_array($model, $this->relatedModels[$cardinality][$model->modelName][$ref], TRUE);
  550.             }
  551.             else {
  552.                 foreach($this->relatedModels[$cardinality][$model->modelName][$ref] as $rModel) {
  553.                     if($model->getPrimaryKeyValue()==$rModel->getPrimaryKeyValue()) {
  554.                         $alreadyInArray = TRUE;
  555.                         break;
  556.                     }
  557.                 }
  558.                 reset($this->relatedModels[$cardinality][$model->modelName][$ref]);
  559.             }
  560.             if(!$alreadyInArray) {
  561.  
  562.                 // If $model already has a defined parent (the 1 in 1:M), and
  563.                 // it's not $this, then we first need to disband that
  564.                 // relationship
  565.                 $foreignKey = $model->getForeignKey($this, $ref);
  566.                 $currentParentModel = $model->getRelatedModelsByCardinality(ModelRelation::MANY_TO_ONE);
  567.                 if(isset($currentParentModel[$foreignKey]) && $currentParentModel[$foreignKey]!==$this) {
  568.                     $currentParentModel[$foreignKey]->removeRelatedModel($model);
  569.                 }
  570.  
  571.                 // Set $model's foreign-key and then add it to $this Model's
  572.                 // list of related Models
  573.                 $model->{$foreignKey} = $this->isInDatabase() ? $this->getPrimaryKeyValue() : -1;
  574.                 $this->relatedModels[$cardinality][$model->modelName][$ref][] = $model;
  575.  
  576.                 // If the relationship has a limit (eg. 1:1 (1:M,1), 1:M,6, etc)
  577.                 // and it has been exceded, then purge the first Model in the
  578.                 // list (ie. the oldest set relation)
  579.                 $limit = $relation->getLimit();
  580.                 if($limit!==NULL && count($this->relatedModels[$cardinality][$model->modelName][$ref])>$limit) {
  581.                     $removedModel = array_shift($this->relatedModels[$cardinality][$model->modelName][$ref]);
  582.                     $removedModel->removeRelatedModels($this);
  583.                 }
  584.  
  585.                 // Add other side of relationship, if bubbling
  586.                 if(!$cancelBubble) {
  587.                     $model->addRelatedModel($this, $ref, NULL, TRUE);
  588.                 }
  589.             }
  590.         }
  591.     }
  592.  
  593.     /*
  594.     # @method void setRelatedModel( Model $model )
  595.     # $model    = The Model to be stored
  596.     #
  597.     # This is an alias for "$this->addRelatedModel()" because sometimes using "set" instead of "add"
  598.     # is better suited, liguistically, to the calling context.
  599.     # Other than that, the two methods are identical.
  600.     */
  601.     public function setRelatedModel($model, $relationRef=ModelRelation::REF_DEFAULT, $reverseRelationRef=NULL) {
  602.  
  603.         // Result
  604.         return $this->addRelatedModel($model, $relationRef, $reverseRelationRef);
  605.     }
  606.  
  607.     /*
  608.     # @method array|Model|NULL getRelatedModels( string $modelName, [string $relationRef] )
  609.     # $modelName    = Specific target Model
  610.     # $relationRef    = Relationship reference
  611.     #
  612.     # Returns all related Models of type $modelName and based on the given
  613.     # relationship reference.
  614.     #
  615.     # This method only returns Models that are currently loaded into
  616.     # "$this->relatedModels" and DOES NOT load anything from persistent storage
  617.     # (see "$this->loadRelatedModels()" for that facility).
  618.     #
  619.     # As with "addRelatedModel()", overloading allows you to use user-friendly
  620.     # method names to retreive certain Models:
  621.     # eg. $A->getClient() .. passes through to .. $A->getRelatedModels('Client')
  622.     #
  623.     # For 1:M,1 and M:1 relations, a single Model instance will be returned, or
  624.     # NULL if there is no related Model.
  625.     #
  626.     # For M:M relationships, an array will be returned containing all related
  627.     # Models or empty if none were found.
  628.     #
  629.     # If $relationRef is NOT specified and only a "__default__" reference exists
  630.     # in the relationship then that reference will be returned.
  631.     # Otherwise an array, indexed by reference code, is returned.
  632.     #
  633.     # Recursive 1:M relations:
  634.     # You MUST specify a $relationRef because otherwise the function can't know
  635.     # which side of the relation you want to retrieve Models from (ie 1 or M).
  636.     # If you don't specify $relationRef, then an array indexed with all
  637.     # relation references will be returned.
  638.     */
  639.     public function getRelatedModels($modelName, $relationRef=NULL) {
  640.  
  641.         // Get the relationship between $this and $modelName
  642.         $relation = ModelRelation::getRelation($this->modelName, $modelName, $relationRef);
  643.  
  644.         // Handle multiple references
  645.         $models = array();
  646.         if(is_array($relation)) {
  647.             if(isset($relation[ModelRelation::REF_DEFAULT]) && count($relation)==1) {
  648.                 $relation = $relation[ModelRelation::REF_DEFAULT];
  649.             }
  650.             else {
  651.                 foreach($relation as $ref=>$rel) {
  652.                     $rModels = $this->getRelatedModels($modelName, $ref);
  653.                     $models[$ref] = $rModels;
  654.                 }
  655.                 return $models;
  656.             }
  657.         }
  658.         $relationRef = $relation->getReference();
  659.  
  660.         // Retrieve all related Models
  661.         switch($relation->getCardinality()) {
  662.  
  663.             // No relationship
  664.             case ModelRelation::NONE:
  665.                 break;
  666.  
  667.             // 1:1
  668.             case ModelRelation::ONE_TO_ONE:
  669.                 break;
  670.  
  671.             // 1:M
  672.             case ModelRelation::ONE_TO_MANY:
  673.                 if(isset($this->relatedModels[ModelRelation::ONE_TO_MANY][$modelName][$relationRef])) {
  674.                     $relatedModels = $this->relatedModels[ModelRelation::ONE_TO_MANY][$modelName][$relationRef];
  675.                     if($relation->getLimit()==1) {
  676.                         if(count($relatedModels)>0) {
  677.                             $models = $relatedModels[0];
  678.                         }
  679.                         else {
  680.                             $models = NULL;
  681.                         }
  682.                     }
  683.                     else {
  684.                         $models = $relatedModels;
  685.                     }
  686.                 }
  687.                 else if($relation->getLimit()==1) {
  688.                     $models = NULL;
  689.                 }
  690.                 break;
  691.  
  692.             // M:1
  693.             case ModelRelation::MANY_TO_ONE:
  694.                 $foreignKey = $this->getForeignKey($modelName, $relationRef);
  695.                 if(isset($this->relatedModels[ModelRelation::MANY_TO_ONE][$foreignKey])) {
  696.                     $models = $this->relatedModels[ModelRelation::MANY_TO_ONE][$foreignKey];
  697.                 }
  698.                 else {
  699.                     $models = NULL;
  700.                 }
  701.                 break;
  702.  
  703.             // M:M
  704.             case ModelRelation::MANY_TO_MANY:
  705.                 if($relation->isRecursive()) {
  706.                     $allLinkModels = $this->getRelatedModels($relation->getLinkModel());
  707.                     foreach($allLinkModels as $ref=>$linkModels) {
  708.                         foreach($linkModels as $linkModel) {
  709.                             $rModels = $linkModel->getRelatedModels($modelName);
  710.                             unset($rModels[$relationRef]);
  711.                             foreach($rModels as $rModel) {
  712.                                 if($rModel!==$this) $models[] = $rModel;
  713.                             }
  714.                         }
  715.                     }
  716.                 }
  717.                 else {
  718.                     $linkModelName = $relation->getLinkModel();
  719.                     if(isset($this->relatedModels[ModelRelation::ONE_TO_MANY][$linkModelName][$relationRef])) {
  720.                         $linkModels = $this->relatedModels[ModelRelation::ONE_TO_MANY][$linkModelName][$relationRef];
  721.                         foreach($linkModels as $linkModel) {
  722.                             $relatedModels = $linkModel->getRelatedModels($modelName, $relationRef);
  723.                             if(!is_array($relatedModels)) {
  724.                                 $models[] = $relatedModels;
  725.                             }
  726.                             else {
  727.                                 $models = array_merge($models, $relatedModels);
  728.                             }
  729.                         }
  730.                     }
  731.                 }
  732.                 break;
  733.  
  734.             // default
  735.             default:
  736.                 break;
  737.         }
  738.  
  739.         // Result
  740.         return $models;
  741.     }
  742.  
  743.     /*
  744.     # @method array getRelatedModelsByCardinality( int $cardinality, [string $modelName] )
  745.     # $cardinality    = Relationship cardinality (with $this on the LEFT of that relationship). Use ModelRelation constants for this value.
  746.     # $modelName    = Specific target Model
  747.     #
  748.     # Returns all related Models with the specified relationship cardinality.
  749.     # For 1:M relationships, you can also specify a target Model in order to return only Models of that type.
  750.     */
  751.     public function getRelatedModelsByCardinality($cardinality, $modelName=NULL) {
  752.  
  753.         // Result
  754.         if($cardinality==ModelRelation::ONE_TO_MANY && !is_null($modelName)) {
  755.             return $this->getRelatedModels($modelName);
  756.         }
  757.         else {
  758.             // Result
  759.             return $this->relatedModels[$cardinality];
  760.         }
  761.     }
  762.  
  763.     /*
  764.     # @method Model|NULL getRelatedModelByInstance( Model $model, [string $relationRef] )
  765.     # $model        = Model instance
  766.     # $relationRef    = Relationship reference to search on
  767.     #
  768.     # Returns an entry in $this->relatedModels that matches $model. This is used to basically get
  769.     # a working reference to an entry in $this->relatedModels.
  770.     #
  771.     # If $relationRef is not specified, then ALL relationships that exist for the Model pair are searched.
  772.     */
  773.     public function getRelatedModelByInstance($model, $relationRef=NULL) {
  774.  
  775.             // Get relation
  776.             $relation = ModelRelation::getRelation($this->modelName, $model->modelName, $relationRef);
  777.  
  778.             // Handle multiple relationship references by calling each one individually to find $model
  779.             if(is_array($relation)) {
  780.                 foreach($relation as $rel) {
  781.                     $rModel = $this->getRelatedModelByInstance($model, $rel->getReference());
  782.                     if(!is_null($rModel)) {
  783.                         break;
  784.                     }
  785.                 }
  786.                 reset($relation);
  787.                 return $rModel;
  788.             }
  789.  
  790.             // Get all related Models
  791.             $relatedModels = $this->getRelatedModels($model->modelName, $relationRef);
  792.             if(is_null($relatedModels)) {
  793.                 return NULL;
  794.             }
  795.             if(!is_array($relatedModels)) {
  796.                 $relatedModels = array($relatedModels);
  797.             }
  798.  
  799.             // Search for $model in $this->relatedModels

  842.     public function removeRelatedModel($model, $relationRef=NULL, $cancelBubble=FALSE) {

  845.         $relation = ModelRelation::getRelation($this->modelName, $model->modelName, $relationRef);
  846.  
  847.         // Handle multiple relationship references by calling each one
  848.         // individually to find $model
  849.         if(is_array($relation)) {
  850.             $processed = array();
  851.             foreach($relation as $rel) {
  852.  
  853.                 // For recursive relations, manually disband both sides of the

  855.                 if($rel->isRecursive()) {

  857.                     // M:M
  858.                     // For M:M relations, we find the model that links $this and

  877.

  879.                 else if(!isset($processed[$rel->getReference()])) {

  885.         }

  888.         switch($relation->getCardinality()) {
  889.  
  890.             // No relationship
  891.             case ModelRelation::NONE:
  892.                 break;

  894.             // 1:M
  895.             case ModelRelation::ONE_TO_MANY:

  902.  
  903.                             // Unset reference in $this->relatedModels
  904.                             unset($this->relatedModels[ModelRelation::ONE_TO_MANY][$model->modelName][$relation->getReference()][$k]);
  905.  
  906.                             // Remove $this from $model
  907.                             if(!$cancelBubble) {
  908.                                 $model->removeRelatedModel($this, $relation->getReference());
  909.                             }
  910.  
  911.                             // Exit loop
  912.                             break;
  913.                         }
  914.                     }
  915.                     reset($this->relatedModels[ModelRelation::ONE_TO_MANY][$model->modelName][$relation->getReference()]);
  916.                 }
  917.  
  918.                 // Break
  919.                 break;
  920.  
  921.             // M:1
  922.             case ModelRelation::MANY_TO_ONE:
  923.  
  924.                 // Set the foreign-key on $this to -1
  925.                 // TODO: -1 is no good for string fields. Use NULL instead? Will
  926.                 // affect other areas so watch out.
  927.                 $foreignKey = $this->getForeignKey($model, $relation->getReference());

  932.

  934.                 unset($this->relatedModels[ModelRelation::MANY_TO_ONE][$foreignKey]);
  935.  
  936.                 // Remove $this from $model
  937.                 if(!$cancelBubble) {

  939.                 }
  940.  
  941.                 // Break
  942.                 break;
  943.  
  944.             // M:M
  945.             case ModelRelation::MANY_TO_MANY:
  946.  
  947.                 // Find a Model instance that links $this and $model
  948.                 $linkModel = $this->getLinkingModel($model);
  949.                 if($linkModel!==NULL) {
  950.                     $linkModel->{$linkModel->getForeignKey($this, $relation->getReference())} = -1;
  951.                     $linkModel->removeRelatedModel($this, $relation->getReference());
  952.                     $linkModel->{$linkModel->getForeignKey($model, $relation->getReference())} = -1;
  953.                     $linkModel->removeRelatedModel($model, $relation->getReference());
  954.                 }
  955.                 break;
  956.  
  957.             // Default
  958.             default:
  959.                 break;
  960.         }
  961.     }

  963.     /*

  1005.     public function loadRelatedModels($targetModelName=NULL, $criteria=NULL, $relationRef=NULL) {
  1006.  
  1007.         // Get the relationship between $this and $targetModelName
  1008.         $relation = ModelRelation::getRelation($this->modelName, $targetModelName, $relationRef);
  1009.  
  1010.         if(!is_null($targetModelName)) {
  1011.             if(is_array($relation)) {
  1012.                 $result = TRUE;
  1013.                 foreach($relation as $rel) {
  1014.                     $result = $result && $this->loadRelatedModels($targetModelName, $criteria, $rel->getReference());
  1015.                 }
  1016.                 return $result;
  1017.             }
  1018.             else {
  1019.                 $relation = array($relation);
  1020.             }
  1021.         }
  1022.  
  1023.         // Load all related Models into $this
  1024.         foreach($relation as $rel) {
  1025.  
  1026.             // Check 'hasLoadedRelatedModels' flag first
  1027.             if(isset($this->hasLoadedRelatedModels[$rel->getTargetModel().$rel->getReference()])) {
  1028.                 continue;
  1029.             }
  1030.  
  1031.             // Handle multiple relationship references
  1032.             if(is_array($rel)) {
  1033.                 $result = TRUE;
  1034.                 foreach($rel as $r) {
  1035.                     $result = $result && $this->loadRelatedModels($targetModelName, $criteria, $r->getReference());
  1036.                 }
  1037.                 return $result;
  1038.             }
  1039.  
  1040.             // Retrieve Models based on their relationship cardinality
  1041.             switch($rel->getCardinality()) {
  1042.     
  1043.                 // No relationship
  1044.                 case ModelRelation::NONE:
  1045.                     break;
  1046.     
  1047.                 // 1:1
  1048.                 case ModelRelation::ONE_TO_ONE:
  1049.                     break;
  1050.     
  1051.                 // 1:M
  1052.                 case ModelRelation::ONE_TO_MANY:
  1053.  
  1054.                     // TODO: If this is part of a recursive M:M then we need to load both the 1:M and M:1 sides of the relationship
  1055.  
  1056.                     // Don't bother finding relations for a non-persistent Model, because it's primary key won't yet be set
  1057.                     // so no foreign-keys can latch onto it in order to build relationships.
  1058.                     if($this->isInDatabase()) {
  1059.  
  1060.                         // Load Models
  1061.                         $tModelName = $rel->getTargetModel();
  1062.                         $tModel = Model::create($tModelName);
  1063.                         //$tManager = ModelManager::get($tModelName);
  1064.                         if(is_null($criteria)) {
  1065.                             $C = new ModelCriteria();
  1066.                             $C->addClause(ModelCriteria::EQUALS, $tModel->getForeignKey($this, $rel->getReference()), $this->getPrimaryKeyValue());
  1067.                             $C->addOrder($tModel->getPrimaryKey(), 'ASC');
  1068.                             $rModels = ModelManager::select($tModelName, $C);
  1069.                         }
  1070.                         else {
  1071.                             $criteria->addClause(ModelCriteria::EQUALS, $tModel->getForeignKey($this, $rel->getReference()), $this->getPrimaryKeyValue());
  1072.                             $rModels = ModelManager::select($tModelName, $criteria);
  1073.                         }
  1074.                         if($rel->getLimit()==1 && count($rModels)>0) {
  1075.                             $this->addRelatedModel($rModels[0], $rel->getReference());
  1076.                         }
  1077.                         else {
  1078.                             foreach($rModels as $rModel) {
  1079.                                 $this->addRelatedModel($rModel, $rel->getReference());
  1080.                             }
  1081.                         }
  1082.                     }
  1083.  
  1084.                     // If the relationship is recursive, then follow through (ie. go straight to M:1 switch case) to handle the M:1 side of the relationship
  1085.                     if($rel->isRecursive()) {
  1086.                         $rel = $rel->getInverseRelation();
  1087.                     }
  1088.                     else {
  1089.                         break;
  1090.                     }
  1091.     
  1092.                 // M:1
  1093.                 case ModelRelation::MANY_TO_ONE:
  1094.  
  1095.                     $tModelName = $rel->getTargetModel();
  1096.                     $tModel = Model::create($tModelName);
  1097.                     $sForeignKey = $this->getForeignKey($tModelName, $rel->getReference());
  1098.                     if(is_null($criteria)) {
  1099.                         $C = new ModelCriteria();
  1100.                         $C->addClause(ModelCriteria::EQUALS, $tModel->getPrimaryKey(), $this->{$sForeignKey});
  1101.                         $C->addOrder($tModel->getPrimaryKey(), 'ASC');
  1102.                         $rModels = ModelManager::select($tModelName, $C);
  1103.                     }
  1104.                     else {
  1105.                         $criteria->addClause(ModelCriteria::EQUALS, $tModel->getPrimaryKey(), $this->{$sForeignKey});
  1106.                         $rModels = ModelManager::select($tModelName, $criteria);
  1107.                     }
  1108.                     if(count($rModels)>0) {
  1109.  
  1110.                         // If the relationship is recursive, then remember we can only
  1111.                         // add recursive related Models via 1:M relationships, so we have
  1112.                         // to reverse the way we add a related Model here.
  1113.                         if($rel->isRecursive()) {
  1114.                             $rModels[0]->addRelatedModel($this, $rel->getInverseRelation()->getReference());
  1115.                         }
  1116.                         else {
  1117.                             $this->addRelatedModel($rModels[0], $rel->getReference());
  1118.                         }
  1119.                     }
  1120.                     break;
  1121.     
  1122.                 // M:M
  1123.                 case ModelRelation::MANY_TO_MANY:
  1124.  
  1125.                     // Get all related instances of the linking Model
  1126.                     $lModelName = $rel->getLinkModel();
  1127.                     $lModel = Model::create($lModelName);
  1128.                     if(is_null($criteria)) {
  1129.                         $C = new ModelCriteria();
  1130.                         $C->addClause(ModelCriteria::EQUALS, $lModel->getForeignKey($this, $rel->getReference()), $this->getPrimaryKeyValue());
  1131.                         $C->addOrder($lModel->getPrimaryKey(), 'ASC');
  1132.                         $linkModels = ModelManager::select($lModelName, $C);
  1133.                     }
  1134.                     else {
  1135.                         $criteria->addClause(ModelCriteria::EQUALS, $lModel->getForeignKey($this, $rel->getReference()), $this->getPrimaryKeyValue());
  1136.                         $linkModels = ModelManager::select($lModelName, $criteria);
  1137.                     }
  1138.  
  1139.                     // Move all Models from $this->relatedModels that match entries
  1140.                     // in $linkModels into $linkModels.
  1141.                     foreach($linkModels as $k=>$linkModel) {
  1142.                         $relatedModel = $this->getRelatedModelByInstance($linkModel);
  1143.                         if(!is_null($relatedModel)) {
  1144.                             $linkModels[$k] = $relatedModel;
  1145.                         }
  1146.                     }
  1147.  
  1148.                     // Get all instances of target model
  1149.                     // If the relationship is recursive then we need to get
  1150.                     // models on the reverse side of the relationship, ie.
  1151.                     // REF_PARENT/REF_CHILD.
  1152.                     $ref = $rel->isRecursive() ? $rel->getInverseRelation()->getReference() : $rel->getReference();
  1153.                     $tModelName = $rel->getTargetModel();
  1154.                     $tModelManager = ModelManager::create($tModelName);
  1155.                     $lForeignKey = $lModel->getForeignKey($tModelName, $ref);
  1156.                     foreach($linkModels as $linkModel) {
  1157.                         $tModel = Model::create($tModelName);
  1158.                         $tModel->setPrimaryKeyValue($linkModel->{$lForeignKey});
  1159.                         if($tModelManager->load($tModel)) {
  1160.                             // TODO: Check that $tModel is not already loaded into $linkModel, and $linkModel is not already loaded into $this
  1161.                             $linkModel->addRelatedModel($tModel, $ref);
  1162.                             $this->addRelatedModel($linkModel, $rel->getReference());
  1163.                         }
  1164.                         unset($tModel);
  1165.                     }
  1166.                     break;
